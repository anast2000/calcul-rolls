function calcul(){ let type = document.getElementById("type").value.split(","); let longueurTotale = parseFloat(type[0]); let largeur = parseFloat(type[1]); let longueurRestante = longueurTotale; let totalGlobal = 0; let tbody = document.querySelector("#detail tbody"); tbody.innerHTML = ""; document.querySelectorAll(".ref").forEach((ref, index) => { if(longueurRestante <= 0) return; let nom = ref.querySelector(".nomRef").value || ("R"+(index+1)); let d = parseFloat(ref.querySelector(".diametre").value) / 100; let lr = parseFloat(ref.querySelector(".largeurRoll").value) / 100; let nbLargeur = Math.floor(largeur / d); let rollsParBloc = 0; let typeBloc = ""; if(nbLargeur == 2){ rollsParBloc = 5; typeBloc = "Pyramide"; } else if(nbLargeur >= 3){ rollsParBloc = 8; typeBloc = "Pile"; } else { rollsParBloc = nbLargeur; typeBloc = "Simple"; } let nbLongueurMax = Math.floor(longueurRestante / lr); let total = nbLongueurMax * rollsParBloc; totalGlobal += total; longueurRestante -= nbLongueurMax * lr; let row = ` ${nom} ${(d*100).toFixed(0)} cm ${typeBloc} (${rollsParBloc}) ${nbLongueurMax} ${total} `; tbody.innerHTML += row; }); document.getElementById("resultat").innerText = "✅ Total réel = " + totalGlobal + " rolls"; document.getElementById("detail").style.display = "table"; }
